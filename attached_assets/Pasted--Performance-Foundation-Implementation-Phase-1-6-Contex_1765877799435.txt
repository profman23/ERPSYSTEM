## TASK 3: Add Slow Query Logging

**Create new file**: `server/src/db/queryLogger.ts`

**Requirements**:
```typescript
import { contextLogger } from '@/core/context/contextLogger';

const SLOW_QUERY_THRESHOLD = 100; // milliseconds

export async function loggedQuery(
  name: string,
  fn: () => Promise
): Promise {
  const start = Date.now();
  
  try {
    const result = await fn();
    const duration = Date.now() - start;
    
    if (duration > SLOW_QUERY_THRESHOLD) {
      contextLogger.warn('slow_query', {
        query: name,
        duration_ms: duration,
        threshold_ms: SLOW_QUERY_THRESHOLD
      });
    }
    
    return result;
  } catch (error) {
    const duration = Date.now() - start;
    contextLogger.error('query_error', {
      query: name,
      duration_ms: duration,
      error: error instanceof Error ? error.message : String(error)
    });
    throw error;
  }
}
```

**Apply to critical queries**: Wrap 5-10 most frequent database queries with `loggedQuery()`

**Example usage**:
```typescript
// Before:
const users = await db.select().from(usersTable);

// After:
const users = await loggedQuery('get_all_users', () =>
  db.select().from(usersTable)
);
```

## TASK 4: Socket.IO Redis Adapter

**File to modify**: `server/src/realtime/socketServer.ts`

**Requirements**:
1. Import Redis adapter: `import { createAdapter } from '@socket.io/redis-adapter';`
2. Check if `redisClient` is available
3. If available, configure Redis adapter for horizontal scaling
4. If not available, log warning and use in-memory adapter

**Implementation**:
```typescript
import { createAdapter } from '@socket.io/redis-adapter';
import { redisClient } from '@/services/redisClient';
import { logger } from '@/config/logger';

export function initializeSocketIO(server: any) {
  const io = new Server(server, {
    cors: { /* existing config */ }
  });

  // Redis adapter for horizontal scaling
  if (redisClient) {
    const pubClient = redisClient.duplicate();
    const subClient = redisClient.duplicate();

    Promise.all([
      pubClient.connect(), 
      subClient.connect()
    ]).then(() => {
      io.adapter(createAdapter(pubClient, subClient));
      logger.info('✅ Socket.IO using Redis adapter');
    }).catch((err) => {
      logger.error('❌ Socket.IO Redis adapter failed:', err);
      logger.warn('⚠️ Falling back to in-memory adapter');
    });
  } else {
    logger.warn('⚠️ Socket.IO using in-memory adapter (no Redis)');
  }

  return io;
}
```

**Expected log**:
```
✅ Socket.IO using Redis adapter
```
